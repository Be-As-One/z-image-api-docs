## Endpoint

```
POST https://z-image.vip/api/v1/jobs/createTask
```

## Headers

| Header | Value | Required |
|--------|-------|----------|
| Authorization | Bearer YOUR_API_KEY | Yes |
| Content-Type | application/json | Yes |

## Request Body

<ParamField body="model" type="string" required>
  Must be `z-image-turbo`
</ParamField>

<ParamField body="inputs" type="object" required>
  Generation parameters

  <Expandable title="inputs properties">
    <ParamField body="prompt" type="string" required>
      Text description of the image to generate (1-500 characters)
    </ParamField>

    <ParamField body="aspectRatio" type="string" default="16:9">
      Output aspect ratio. Options: `16:9`, `1:1`, `9:16`, `4:3`, `3:4`, `21:9`
    </ParamField>

    <ParamField body="resolution" type="string" default="2k">
      Output quality. Options:
      - `2k` - ~2048px, 5 credits
      - `4k` - ~4096px, 8 credits
    </ParamField>

    <ParamField body="negativePrompt" type="string">
      Elements to exclude (optional, max 200 characters)
    </ParamField>
  </Expandable>
</ParamField>

## Response

<ResponseField name="code" type="number">
  200 for success
</ResponseField>

<ResponseField name="message" type="string">
  "success" on success
</ResponseField>

<ResponseField name="data" type="object">
  <Expandable title="data properties">
    <ResponseField name="taskId" type="string">
      Unique task identifier for polling status
    </ResponseField>

    <ResponseField name="orderId" type="string">
      Order identifier
    </ResponseField>

    <ResponseField name="creditsUsed" type="number">
      Credits deducted for this task (5 or 8)
    </ResponseField>
  </Expandable>
</ResponseField>

## Example Request

<CodeGroup>

```javascript Node.js
const response = await fetch('https://z-image.vip/api/v1/jobs/createTask', {
  method: 'POST',
  headers: {
    'Authorization': 'Bearer YOUR_API_KEY',
    'Content-Type': 'application/json'
  },
  body: JSON.stringify({
    model: 'z-image-turbo',
    inputs: {
      prompt: 'A serene mountain landscape at golden hour',
      aspectRatio: '16:9',
      resolution: '2k',
      negativePrompt: 'blurry, low quality'
    }
  })
});

const data = await response.json();
console.log(data.data.taskId);
```

```python Python
import requests

response = requests.post(
    'https://z-image.vip/api/v1/jobs/createTask',
    headers={
        'Authorization': 'Bearer YOUR_API_KEY',
        'Content-Type': 'application/json'
    },
    json={
        'model': 'z-image-turbo',
        'inputs': {
            'prompt': 'A serene mountain landscape at golden hour',
            'aspectRatio': '16:9',
            'resolution': '2k',
            'negativePrompt': 'blurry, low quality'
        }
    }
)

data = response.json()
print(data['data']['taskId'])
```

```bash cURL
curl -X POST https://z-image.vip/api/v1/jobs/createTask \
  -H "Authorization: Bearer YOUR_API_KEY" \
  -H "Content-Type: application/json" \
  -d '{
    "model": "z-image-turbo",
    "inputs": {
      "prompt": "A serene mountain landscape at golden hour",
      "aspectRatio": "16:9",
      "resolution": "2k",
      "negativePrompt": "blurry, low quality"
    }
  }'
```
